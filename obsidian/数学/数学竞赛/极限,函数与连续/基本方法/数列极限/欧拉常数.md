$$
\begin{align}
\lim_{ n \to \infty } \left( \sum_{k=1}^n \frac{1}{k}-\ln n \right)=C=0.57721\cdots
\end{align}
$$
```python
import numpy as np  
import matplotlib.pyplot as plt  
  
n=np.arange(1e2,1e4)  
def sum_x(n1):  
  H_n1=np.array([np.sum(1.0/np.arange(1,k+1)) for k in n1])  
  return H_n1  
C=sum_x(n)-np.log(n)  
plt.plot(n,C)  
plt.axhline(0.577215,color='red',label='0.577215')  
plt.legend()  
plt.show()
```
