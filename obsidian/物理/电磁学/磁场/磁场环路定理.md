$$
\begin{align}
\oint_{L_{0}} \mathbf{B}\cdot d\mathbf{l}=\mu_{0} \sum I_{i}\\[10pt]
\nabla \times \mathbf{B}=\mu_{0}\,\mathbf{j}
\end{align}
$$
